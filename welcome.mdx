## Quick Start Examples

<Tabs>
  <Tab title="JavaScript">
    ```javascript
    import { Cobalt } from '@cobaltio/cobalt-js';

    const cobalt = new Cobalt({
      apiKey: 'your-api-key'
    });

    // Create a linked account for your user
    const linkedAccount = await cobalt.linkedAccounts.create({
      customerId: 'user-123',
      name: 'John Doe'
    });

    // Enable Salesforce integration
    const integration = await cobalt.integrations.enable({
      linkedAccountId: linkedAccount.id,
      slug: 'salesforce'
    });

    console.log('Integration ready:', integration.status);
    ```
  </Tab>
  <Tab title="Python">
    ```python
    from cobalt import Cobalt

    cobalt = Cobalt(api_key='your-api-key')

    # Create a linked account for your user
    linked_account = cobalt.linked_accounts.create(
        customer_id='user-123',
        name='John Doe'
    )

    # Enable Salesforce integration
    integration = cobalt.integrations.enable(
        linked_account_id=linked_account.id,
        slug='salesforce'
    )

    print(f'Integration ready: {integration.status}')
    ```
  </Tab>
  <Tab title="cURL">
    ```bash
    # Create a linked account
    curl -X POST https://api.gocobalt.io/v1/linked-accounts \
      -H "Authorization: Bearer your-api-key" \
      -H "Content-Type: application/json" \
      -d '{
        "customer_id": "user-123",
        "name": "John Doe"
      }'

    # Enable Salesforce integration
    curl -X POST https://api.gocobalt.io/v1/integrations \
      -H "Authorization: Bearer your-api-key" \
      -H "Content-Type: application/json" \
      -d '{
        "linked_account_id": "la_123",
        "slug": "salesforce"
      }'
    ```
  </Tab>
</Tabs>
