---
title: Get Workflow Draft
api: get /api/v2/public/workflow/:workflow_id/edit
description: Create a draft version of a public workflow for editing.
---

Create a draft (editable) version of a public workflow. Modifications are made in draft mode before being published. This enables users to test changes before making them live.

<Info>
Draft workflows persist until they are published, allowing users to return and modify them anytime before finalizing. The live workflow remains unchanged until the draft is published.
</Info>

## Authentication

<ParamField header="Authorization" type="string" required>
  Bearer token using your session token for authentication.
  
  **Format:** `Bearer YOUR_SESSION_TOKEN`
  
  Generate a session token using the [Session Token API](/v3/api-reference/session-token/generate-session-token).
</ParamField>

## Path Parameters

<ParamField path="workflow_id" type="string" required>
  The unique identifier of the workflow to create a draft for.
  
  **Example:** `67dd309c8a29c6f92d8bb45e`
</ParamField>

## Example: Get Workflow Draft
```javascript
async function getWorkflowDraft(sessionToken, workflowId) {
  const response = await fetch(
    `https://api.gocobalt.io/api/v2/public/workflow/${workflowId}/edit`,
    {
      headers: {
        'Authorization': `Bearer ${sessionToken}`
      }
    }
  );

  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.message);
  }

  return response.json();
}

// Usage: Create draft for editing
const draft = await getWorkflowDraft('session_token_here', '67dd309c8a29c6f92d8bb45e');

console.log(`Draft created: ${draft._id}`);
console.log(`Original workflow: ${draft.associated_public_workflow}`);
console.log(`Nodes: ${draft.nodes.length}`);
```

## Example: Edit and Embed Draft Workflow
```javascript
async function openWorkflowEditor(workflowId, sessionToken, containerId) {
  // Get or create draft
  const draft = await getWorkflowDraft(sessionToken, workflowId);
  
  // Embed the workflow editor
  const iframe = document.createElement('iframe');
  iframe.src = `https://app.gocobalt.io/workflows/${draft._id}?token=${sessionToken}`;
  iframe.style.width = '100%';
  iframe.style.height = '600px';
  iframe.style.border = 'none';
  
  const container = document.getElementById(containerId);
  container.innerHTML = '';
  container.appendChild(iframe);
  
  return draft;
}

// Usage
const draft = await openWorkflowEditor(
  '67dd309c8a29c6f92d8bb45e',
  'session_token_here',
  'workflow-editor-container'
);
```

## Example: Workflow Edit Flow
```javascript
async function workflowEditFlow(sessionToken, workflowId) {
  // 1. Get the draft
  console.log('Creating draft...');
  const draft = await getWorkflowDraft(sessionToken, workflowId);
  console.log(`Draft ID: ${draft._id}`);

  // 2. Display editor URL for the user
  const editorUrl = `https://app.gocobalt.io/workflows/${draft._id}?token=${sessionToken}`;
  console.log(`\nEdit your workflow at:\n${editorUrl}`);

  // 3. Provide instructions
  console.log('\nWorkflow Editing Steps:');
  console.log('1. Make changes in the editor');
  console.log('2. Test changes using Workflow Testing');
  console.log('3. Click "Publish" to make changes live');
  console.log('4. Or click "Cancel" to save as draft for later');

  return {
    draftId: draft._id,
    originalId: draft.associated_public_workflow,
    editorUrl
  };
}
```
