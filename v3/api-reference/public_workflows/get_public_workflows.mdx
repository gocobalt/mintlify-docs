---
title: List Workflows
api: get /api/v2/public/workflow
description: Retrieve workflows with filtering by application, name, and publish status.
---

Retrieve a paginated list of workflows with optional filtering. This API supports both public workflows (organization-wide) and private workflows (linked account specific).

<Info>
When `linked_account_id` is provided, the API returns private workflows created by that linked account. When omitted, only public workflows are returned.
</Info>

## Authentication

<ParamField header="x-api-key" type="string" required>
  Your Refold API key. Find it in **Settings → Credentials**.
</ParamField>

<ParamField header="linked_account_id" type="string">
  The unique identifier for the linked account.
  
  - **If provided:** Returns private workflows created by that linked account
  - **If omitted:** Returns only public (organization-wide) workflows
</ParamField>

## Query Parameters

<ParamField query="slug" type="string">
  Filter workflows by application slug.
  
  **Example:** `hubspot`, `salesforce`, `slack`
</ParamField>

<ParamField query="name" type="string">
  Search workflows by name (case-insensitive partial match).
  
  **Example:** `create`, `sync`, `delete`
</ParamField>

<ParamField query="published" type="boolean">
  Filter by publish status.
  
  - `true` - Only published workflows
  - `false` - Only unpublished (draft) workflows
</ParamField>

<ParamField query="page" type="integer" default="1">
  Page number for pagination.
</ParamField>

<ParamField query="limit" type="integer" default="10">
  Number of records per page (maximum 100).
</ParamField>

## Example: List All Workflows
```javascript
async function listWorkflows(apiKey, options = {}) {
  const params = new URLSearchParams();
  
  if (options.slug) params.append('slug', options.slug);
  if (options.name) params.append('name', options.name);
  if (options.published !== undefined) params.append('published', options.published);
  if (options.page) params.append('page', options.page);
  if (options.limit) params.append('limit', options.limit);

  const url = `https://api.gocobalt.io/api/v2/public/workflow${params.toString() ? '?' + params : ''}`;
  
  const headers = {
    'x-api-key': apiKey
  };
  
  // Include linked_account_id for private workflows
  if (options.linkedAccountId) {
    headers['linked_account_id'] = options.linkedAccountId;
  }

  const response = await fetch(url, { headers });

  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.message);
  }

  return response.json();
}

// Usage: Get all public HubSpot workflows
const result = await listWorkflows('your_api_key', {
  slug: 'hubspot'
});

console.log(`Found ${result.totalDocs} workflows`);
result.docs.forEach(w => {
  console.log(`- ${w.name} (${w.published ? 'Published' : 'Draft'})`);
});
```

## Example: Search Workflows by Name
```javascript
async function searchWorkflows(apiKey, searchTerm, options = {}) {
  const result = await listWorkflows(apiKey, {
    name: searchTerm,
    ...options
  });

  return result.docs;
}

// Usage: Find all "create" workflows for HubSpot
const createWorkflows = await searchWorkflows('your_api_key', 'create', {
  slug: 'hubspot',
  published: true
});

console.log(`Found ${createWorkflows.length} workflows matching "create":`);
createWorkflows.forEach(w => console.log(`- ${w.name}`));
```

## Example: Get Private Workflows for a User
```javascript
async function getUserWorkflows(apiKey, linkedAccountId, slug) {
  const result = await listWorkflows(apiKey, {
    slug,
    linkedAccountId // This fetches private workflows
  });

  return {
    published: result.docs.filter(w => w.published),
    drafts: result.docs.filter(w => !w.published),
    total: result.totalDocs
  };
}

// Usage: Get user's HubSpot workflows
const userWorkflows = await getUserWorkflows(
  'your_api_key',
  'user_12345',
  'hubspot'
);

console.log(`Published: ${userWorkflows.published.length}`);
console.log(`Drafts: ${userWorkflows.drafts.length}`);
```

## Example: Paginate Through All Workflows
```javascript
async function getAllWorkflows(apiKey, options = {}) {
  const allWorkflows = [];
  let page = 1;
  let hasMore = true;

  while (hasMore) {
    const result = await listWorkflows(apiKey, {
      ...options,
      page,
      limit: 100
    });

    allWorkflows.push(...result.docs);
    hasMore = result.hasNextPage;
    page++;
  }

  return allWorkflows;
}

// Usage: Get all published workflows
const allPublished = await getAllWorkflows('your_api_key', {
  published: true
});

console.log(`Total published workflows: ${allPublished.length}`);
```

## Example: Group Workflows by Application
```javascript
async function getWorkflowsByApplication(apiKey) {
  const allWorkflows = await getAllWorkflows(apiKey);

  // Group by slug
  const byApp = allWorkflows.reduce((acc, workflow) => {
    const slug = workflow.slug || 'unknown';
    if (!acc[slug]) acc[slug] = [];
    acc[slug].push(workflow);
    return acc;
  }, {});

  return byApp;
}

// Usage
const grouped = await getWorkflowsByApplication('your_api_key');

Object.entries(grouped).forEach(([app, workflows]) => {
  console.log(`\n${app}: ${workflows.length} workflows`);
  workflows.forEach(w => console.log(`  - ${w.name}`));
});
```
