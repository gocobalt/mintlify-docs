---
title: Get Action Schema
api: get /api/v2/integration-schema/:slug/actions/:action_id
description: Retrieve the parameter schema for a specific action.
---

Retrieve the parameter schema for a specific action within an integration application. Use this to understand what parameters are required or optional before executing an action.

## Authentication

This API supports two authentication methods. Use either Session Token OR API Key with Linked Account ID.

<Tabs>
  <Tab title="Session Token">
    <ParamField header="Authorization" type="string" required>
      Bearer token for authentication.
      
      **Format:** `Bearer YOUR_SESSION_TOKEN`
    </ParamField>
  </Tab>
  <Tab title="API Key + Linked Account">
    <ParamField header="x-api-key" type="string" required>
      Your Refold API key. Find it in **Settings → Credentials**.
    </ParamField>
    
    <ParamField header="linked_account_id" type="string" required>
      The unique identifier for the linked account.
      
      **Example:** `user_12345`
    </ParamField>
  </Tab>
</Tabs>

## Path Parameters

<ParamField path="slug" type="string" required>
  The application slug identifier.
  
  **Example:** `hubspot`, `salesforce`, `slack`
</ParamField>

<ParamField path="action_id" type="string" required>
  The unique action identifier obtained from the List Actions endpoint.
  
  **Example:** `create_contact`, `create_deal`, `update_account`
</ParamField>

## Example: Get Action Parameters
```javascript
async function getActionSchema(slug, actionId, linkedAccountId, apiKey) {
  const response = await fetch(
    `https://api.gocobalt.io/api/v2/integration-schema/${slug}/actions/${actionId}`,
    {
      headers: {
        'x-api-key': apiKey,
        'linked_account_id': linkedAccountId
      }
    }
  );

  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.message);
  }

  return response.json();
}

// Usage: Get schema for HubSpot create_contact action
const schema = await getActionSchema('hubspot', 'create_contact', 'user_12345', 'your_api_key');

const requiredFields = schema.filter(p => p.required);
const optionalFields = schema.filter(p => !p.required);

console.log('Required fields:');
requiredFields.forEach(f => console.log(`- ${f.name} (${f.type})`));

console.log('\nOptional fields:');
optionalFields.forEach(f => console.log(`- ${f.name} (${f.type})`));
```

## Example: Build Dynamic Form
```javascript
async function buildActionForm(slug, actionId, linkedAccountId, apiKey) {
  const schema = await getActionSchema(slug, actionId, linkedAccountId, apiKey);

  // Generate form fields based on schema
  const formFields = schema.map(param => ({
    name: param.name,
    label: param.label,
    type: mapParamType(param.type),
    required: param.required,
    options: param.options || [],
    placeholder: param.placeholder || ''
  }));

  return formFields;
}

function mapParamType(schemaType) {
  const typeMap = {
    'text': 'text',
    'number': 'number',
    'json': 'textarea',
    'boolean': 'checkbox',
    'select': 'select'
  };
  return typeMap[schemaType] || 'text';
}

// Usage: Build form for Salesforce create_account
const formFields = await buildActionForm('salesforce', 'create_account', 'user_12345', 'your_api_key');

formFields.forEach(field => {
  console.log(`<input name="${field.name}" type="${field.type}" required="${field.required}" />`);
});
```

## Example: Validate Parameters Before Execution
```javascript
async function validateActionParams(slug, actionId, params, linkedAccountId, apiKey) {
  const schema = await getActionSchema(slug, actionId, linkedAccountId, apiKey);

  const errors = [];
  const requiredFields = schema.filter(p => p.required);

  // Check required fields
  requiredFields.forEach(field => {
    if (!params[field.name]) {
      errors.push(`Missing required field: ${field.label} (${field.name})`);
    }
  });

  // Check field types
  schema.forEach(field => {
    if (params[field.name] !== undefined) {
      if (field.type === 'json' && typeof params[field.name] === 'string') {
        try {
          JSON.parse(params[field.name]);
        } catch {
          errors.push(`Invalid JSON for field: ${field.name}`);
        }
      }
    }
  });

  return {
    valid: errors.length === 0,
    errors
  };
}

// Usage: Validate before executing
const validation = await validateActionParams(
  'hubspot',
  'create_contact',
  { firstname: 'John' }, // Missing required 'email' field
  'user_12345',
  'your_api_key'
);

if (!validation.valid) {
  console.log('Validation errors:', validation.errors);
}
```
