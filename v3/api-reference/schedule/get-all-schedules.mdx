---
title: List Schedules
api: get /api/v1/workflow/public/scheduled
description: Retrieve all scheduled jobs with optional filtering and pagination.
---

Retrieve a paginated list of all scheduled jobs. Filter by linked account to view schedules for specific users.

## Authentication

<ParamField header="x-api-key" type="string" required>
  Your Refold API key. Find it in **Settings → Credentials**.
</ParamField>

## Query Parameters

<ParamField query="linked_account_id" type="string">
  Filter schedules by linked account ID.
</ParamField>

<ParamField query="page" type="integer" default="1">
  Page number for pagination.
</ParamField>

<ParamField query="limit" type="integer" default="10">
  Number of records per page.
</ParamField>

## Example: List All Schedules
```javascript
async function listSchedules(apiKey, options = {}) {
  const params = new URLSearchParams();
  if (options.linkedAccountId) params.append('linked_account_id', options.linkedAccountId);
  if (options.page) params.append('page', options.page);
  if (options.limit) params.append('limit', options.limit);

  const url = `https://api.gocobalt.io/api/v1/workflow/public/scheduled${params.toString() ? '?' + params : ''}`;

  const response = await fetch(url, {
    headers: {
      'x-api-key': apiKey
    }
  });

  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.message);
  }

  return response.json();
}

// Usage: List all schedules for a user
const result = await listSchedules('your_api_key', {
  linkedAccountId: 'user_12345'
});

console.log(`Found ${result.totalDocs} schedules`);
result.docs.forEach(schedule => {
  const status = schedule.inactive ? '⏸️ Paused' : '▶️ Active';
  console.log(`${status} ${schedule.workflow.name} - ${schedule.cron_patten || 'Interval'}`);
});
```

## Example: Get Schedule Statistics
```javascript
async function getScheduleStats(apiKey, linkedAccountId) {
  const result = await listSchedules(apiKey, { linkedAccountId, limit: 100 });

  const stats = {
    total: result.totalDocs,
    active: result.docs.filter(s => !s.inactive).length,
    paused: result.docs.filter(s => s.inactive).length,
    cronSchedules: result.docs.filter(s => s.trigger_type === 'cron').length,
    intervalSchedules: result.docs.filter(s => s.trigger_type === 'interval').length,
    byApplication: {}
  };

  result.docs.forEach(schedule => {
    const appName = schedule.workflow.associated_application?.name || 'Unknown';
    stats.byApplication[appName] = (stats.byApplication[appName] || 0) + 1;
  });

  return stats;
}

// Usage
const stats = await getScheduleStats('your_api_key', 'user_12345');
console.log(`Total: ${stats.total}, Active: ${stats.active}, Paused: ${stats.paused}`);
```
