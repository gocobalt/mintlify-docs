---
title: Get Hosted Properties
api: get /api/v2/public/hosted-properties
description: Retrieve the hosted portal properties saved against a session token.
---

Retrieve the customization properties (color, background color, customer name) that were saved when creating a hosted URL for a session token.

<Note>
This API uses Session Token authentication instead of API Key authentication.
</Note>

## Authentication

<ParamField header="Authorization" type="string" required>
  Session token as Bearer token. Generate a session token using the [Generate Session Token](/v3/api-reference/session-token/generate-session-token) endpoint.
  
  **Format:** `Bearer {session_token}`
</ParamField>

## Example: Get Hosted Properties
```javascript
async function getHostedProperties(sessionToken) {
  const response = await fetch(
    'https://api.gocobalt.io/api/v2/public/hosted-properties',
    {
      headers: {
        'Authorization': `Bearer ${sessionToken}`
      }
    }
  );

  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.message);
  }

  return response.json();
}

// Usage
const properties = await getHostedProperties('your_session_token');
console.log(`Customer: ${properties.name}`);
console.log(`Theme color: ${properties.color}`);
console.log(`Background: ${properties.bgColor}`);
```

## Example: Sync Properties Across Sessions
```javascript
async function syncHostedProperties(apiKey, linkedAccountId) {
  // Step 1: Generate session token
  const tokenResponse = await fetch(
    'https://api.gocobalt.io/api/v2/public/session-token',
    {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-api-key': apiKey
      },
      body: JSON.stringify({
        linked_account_id: linkedAccountId
      })
    }
  );

  const { token: sessionToken } = await tokenResponse.json();

  // Step 2: Get existing properties
  try {
    const properties = await getHostedProperties(sessionToken);
    return {
      sessionToken,
      properties,
      hasExistingProperties: true
    };
  } catch (error) {
    // No properties saved yet
    return {
      sessionToken,
      properties: null,
      hasExistingProperties: false
    };
  }
}

// Usage
const { sessionToken, properties, hasExistingProperties } = 
  await syncHostedProperties('your_api_key', 'user_12345');

if (hasExistingProperties) {
  console.log(`Existing customization found for ${properties.name}`);
} else {
  console.log('No customization saved, using defaults');
}
```

## Example: Apply Properties to Custom UI
```javascript
async function applyHostedPropertiesToUI(sessionToken, containerElement) {
  const properties = await getHostedProperties(sessionToken);

  // Apply colors to your custom integration UI
  containerElement.style.backgroundColor = properties.bgColor;
  containerElement.style.color = properties.color;

  // Display customer name
  const headerElement = containerElement.querySelector('.header-title');
  if (headerElement) {
    headerElement.textContent = `Integrations for ${properties.name}`;
  }

  return properties;
}

// Usage
const container = document.getElementById('integrations-panel');
const properties = await applyHostedPropertiesToUI('your_session_token', container);
```
