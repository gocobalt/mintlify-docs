---
title: List Executions
api: get /api/v2/public/execution
description: Retrieve a paginated list of workflow executions.
---

Retrieve a paginated list of all workflow executions for a linked account. Use this to monitor workflow activity, debug issues, or build execution history views.

## Authentication

<ParamField header="x-api-key" type="string" required>
  Your Refold API key. Find it in **Settings → Credentials**.
</ParamField>

<ParamField header="linked_account_id" type="string" required>
  The unique identifier for the linked account.
</ParamField>

## Query Parameters

<ParamField query="page" type="integer" default="1">
  Page number for pagination.
</ParamField>

<ParamField query="limit" type="integer" default="10">
  Number of records per page (maximum 100).
</ParamField>

## Example: List Executions
```javascript
async function listExecutions(apiKey, linkedAccountId, options = {}) {
  const params = new URLSearchParams();
  if (options.page) params.append('page', options.page);
  if (options.limit) params.append('limit', options.limit);

  const url = `https://api.gocobalt.io/api/v2/public/execution${params.toString() ? '?' + params : ''}`;

  const response = await fetch(url, {
    headers: {
      'x-api-key': apiKey,
      'linked_account_id': linkedAccountId
    }
  });

  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.message);
  }

  return response.json();
}

// Usage
const result = await listExecutions('your_api_key', 'user_12345', {
  page: 1,
  limit: 20
});

console.log(`Found ${result.totalDocs} executions`);
result.docs.forEach(exec => {
  const status = exec.status === 'COMPLETED' ? '✅' : exec.status === 'ERRORED' ? '❌' : '⏳';
  console.log(`${status} ${exec.associated_workflow} - ${exec.createdAt}`);
});
```

## Example: Get Execution Statistics
```javascript
async function getExecutionStats(apiKey, linkedAccountId) {
  // Get all executions
  const allExecutions = [];
  let page = 1;
  let hasMore = true;

  while (hasMore) {
    const result = await listExecutions(apiKey, linkedAccountId, {
      page,
      limit: 100
    });
    allExecutions.push(...result.docs);
    hasMore = result.hasNextPage;
    page++;
  }

  // Calculate statistics
  const stats = {
    total: allExecutions.length,
    completed: allExecutions.filter(e => e.status === 'COMPLETED').length,
    running: allExecutions.filter(e => e.status === 'RUNNING').length,
    errored: allExecutions.filter(e => e.status === 'ERRORED').length,
    byApplication: {},
    byWorkflow: {}
  };

  // Group by application
  allExecutions.forEach(exec => {
    const app = exec.associated_application;
    stats.byApplication[app] = (stats.byApplication[app] || 0) + 1;
  });

  // Group by workflow
  allExecutions.forEach(exec => {
    const workflow = exec.associated_workflow;
    stats.byWorkflow[workflow] = (stats.byWorkflow[workflow] || 0) + 1;
  });

  // Calculate success rate
  stats.successRate = stats.total > 0 
    ? ((stats.completed / stats.total) * 100).toFixed(1) + '%'
    : 'N/A';

  return stats;
}

// Usage
const stats = await getExecutionStats('your_api_key', 'user_12345');
console.log(`Total: ${stats.total}`);
console.log(`Success Rate: ${stats.successRate}`);
console.log(`Completed: ${stats.completed}, Running: ${stats.running}, Errored: ${stats.errored}`);
```

## Example: Filter Executions by Status
```javascript
async function getExecutionsByStatus(apiKey, linkedAccountId, status) {
  const result = await listExecutions(apiKey, linkedAccountId, { limit: 100 });
  return result.docs.filter(exec => exec.status === status);
}

// Usage: Get all errored executions
const erroredExecutions = await getExecutionsByStatus(
  'your_api_key',
  'user_12345',
  'ERRORED'
);

console.log(`Found ${erroredExecutions.length} errored executions:`);
erroredExecutions.forEach(exec => {
  console.log(`- ${exec.associated_workflow} at ${exec.createdAt}`);
});
```

## Example: Build Execution Dashboard Data
```javascript
async function getDashboardData(apiKey, linkedAccountId) {
  const result = await listExecutions(apiKey, linkedAccountId, { limit: 50 });

  // Recent executions
  const recentExecutions = result.docs.slice(0, 10).map(exec => ({
    id: exec._id,
    workflow: exec.associated_workflow,
    application: exec.associated_application,
    status: exec.status,
    timestamp: new Date(exec.createdAt),
    environment: exec.environment
  }));

  // Status breakdown
  const statusCounts = result.docs.reduce((acc, exec) => {
    acc[exec.status] = (acc[exec.status] || 0) + 1;
    return acc;
  }, {});

  // Application breakdown
  const appCounts = result.docs.reduce((acc, exec) => {
    acc[exec.associated_application] = (acc[exec.associated_application] || 0) + 1;
    return acc;
  }, {});

  return {
    recentExecutions,
    statusCounts,
    appCounts,
    totalExecutions: result.totalDocs,
    pagination: {
      currentPage: result.page,
      totalPages: result.totalPages,
      hasMore: result.hasNextPage
    }
  };
}

// Usage
const dashboard = await getDashboardData('your_api_key', 'user_12345');
console.log('Dashboard Data:', dashboard);
```
