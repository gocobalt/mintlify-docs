---
title: Get Application
api: get /api/v2/public/application/:slug
description: Retrieve details of a specific application by its slug.
---

Retrieve detailed information about a specific application using its slug identifier. Returns connection status, authentication requirements, and configuration details for the linked account.

## Authentication

<ParamField header="x-api-key" type="string" required>
  Your Refold API key. Find it in **Settings → Credentials**.
</ParamField>

<ParamField header="linked_account_id" type="string" required>
  The unique identifier for the linked account.
  
  **Example:** `user_12345`
</ParamField>

## Path Parameters

<ParamField path="slug" type="string" required>
  The application slug identifier.
  
  **Example:** `slack`, `salesforce`, `hubspot`
</ParamField>

## Example: Check Connection Status
```javascript
async function getApplication(slug, linkedAccountId, apiKey) {
  const response = await fetch(
    `https://api.gocobalt.io/api/v2/public/application/${slug}`,
    {
      headers: {
        'x-api-key': apiKey,
        'linked_account_id': linkedAccountId
      }
    }
  );

  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.message);
  }

  return response.json();
}

// Usage: Check if Salesforce is connected
const salesforce = await getApplication('salesforce', 'user_12345', 'your_api_key');

if (salesforce.connected) {
  console.log(`Salesforce connected as: ${salesforce.connected_accounts[0].identifier}`);
  console.log(`Connected at: ${salesforce.connected_accounts[0].connectedAt}`);
} else {
  console.log('Salesforce is not connected');
}

if (salesforce.reauth_required) {
  console.log('Warning: Re-authentication required');
}
```

## Example: Build Auth Form for Key-Based Apps
```javascript
async function getAuthFields(slug, linkedAccountId, apiKey) {
  const app = await getApplication(slug, linkedAccountId, apiKey);
  
  if (app.auth_type !== 'keybased') {
    return { type: 'oauth', app };
  }

  // Return fields needed for authentication form
  return {
    type: 'keybased',
    app,
    fields: app.auth_input_map.map(field => ({
      name: field.name,
      label: field.label,
      placeholder: field.placeholder,
      required: field.required,
      inputType: field.type,
      options: field.options || []
    }))
  };
}

// Usage: Get auth requirements for Copper CRM
const authInfo = await getAuthFields('copper', 'user_12345', 'your_api_key');

if (authInfo.type === 'keybased') {
  console.log('Required fields:');
  authInfo.fields.forEach(f => {
    console.log(`- ${f.label} (${f.required ? 'required' : 'optional'})`);
  });
}
```
