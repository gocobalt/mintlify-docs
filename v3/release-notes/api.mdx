```tsx
// New React SDK usage
import { useCobalt, CobaltProvider } from '@cobalt/react';

function App() {
  return (
    <CobaltProvider apiKey={process.env.REACT_APP_COBALT_API_KEY}>
      <IntegrationDashboard />
    </CobaltProvider>
  );
}

function IntegrationDashboard() {
  const { linkedAccounts, isLoading } = useCobalt();
  
  if (isLoading) return <div>Loading...</div>;
  
  return (
    <div>
      {linkedAccounts.map(account => (
        <div key={account.id}>{account.name}</div>
      ))}
    </div>
  );
}
```
